onde esta meuserviço editar com nome 

sudo vim /etc/systemd/system/meuservico.service 


[Unit]
Description=Meu Serviço Python
After=network.target

[Service]
ExecStart=/usr/bin/python3 /home/robotics/Desktop/door/door.py
WorkingDirectory=/home/robotics/Desktop/door/
StandardOutput=inherit
StandardError=inherit
Restart=always
User=pi  # Certifique-se que esse usuário é válido e tem permissões

[Install]
WantedBy=multi-user.target


sudo systemctl daemon-reload


sudo systemctl enable meuservico.service


sudo systemctl start meuservico.service



sudo systemctl status meuservico.service


sudo systemctl stop meuservico.service


sudo systemctl status meuservico.service
